    4. Create `src/components/AppShowcaseClient.tsx` — **client component ('use client')**:
       ```typescript
       'use client';

       import { motion, useReducedMotion } from 'motion/react';
       import { FadeInSection, staggerContainerVariants, staggerItemVariants } from '@/components/ui/FadeInSection';
       import { IPhoneFrame } from '@/components/IPhoneFrame';
       import { AppScreenshot } from '@/components/AppScreenshot';

       const SCREENSHOTS = [
         { slug: 'iphone-home', alt: 'BitRemote app home screen showing active downloads', width: 390, height: 844 },
       ] as const;

       export function AppShowcaseClient() {
         const shouldReduceMotion = useReducedMotion();

         return (
           <FadeInSection as="section" className="py-12 lg:py-16">
             <h2 className="m-0 mb-8 text-center font-sans text-xl font-semibold">
               See it in action
             </h2>
             <motion.div
               className="flex flex-wrap justify-center gap-6"
               variants={staggerContainerVariants}
               initial={shouldReduceMotion ? 'visible' : 'hidden'}
               whileInView="visible"
               viewport={{ once: true, amount: 0.08 }}
             >
               {SCREENSHOTS.map((shot) => (
                 <motion.div key={shot.slug} variants={staggerItemVariants}>
                   <IPhoneFrame>
                     <AppScreenshot
                       slug={shot.slug}
                       alt={shot.alt}
                       width={shot.width}
                       height={shot.height}
                     />
                   </IPhoneFrame>
                 </motion.div>
               ))}
             </motion.div>
           </FadeInSection>
         );
       }
       ```
       NOTE: Since `motion.div` requires `'use client'`, and `IPhoneFrame`/`AppScreenshot` are server components used inside a client component, Next.js 15 handles this correctly — server components imported into client components are still rendered on the server and passed as props/children. The `SCREENSHOTS` array is duplicated here (same data) because the client component needs it for the map. If later this feels redundant, it can be extracted to a shared data file, but for now keeping it simple.

       IMPORTANT: Use `useReducedMotion` from `motion/react` (same pattern as FadeInSection.tsx) to set `initial` to `'visible'` when reduced motion is preferred — this ensures animations are completely absent, not just skipped.
